Definition

species-prod :  {l1 l2 : Level} (F : species l1) (G : species l2) (X : 𝔽) → UU (l1 ⊔ l2)
species-prod F G X = (F X) × (G X)

_×ˢ_,_ :  {l1 l2 : Level} (F : species l1) (G : species l2) (X : 𝔽) → UU (l1 ⊔ l2)
F ×ˢ G , X = species-prod F G X
