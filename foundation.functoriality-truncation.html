Idea

The universal property of truncations can be used to define the functorial action of truncations.

Definition

unique-map-trunc :
  {l1 l2 : Level} {A : UU l1} {B : UU l2} (k : ùïã) (f : A ‚Üí B) ‚Üí
  is-contr
    ( Œ£ ( type-trunc k A ‚Üí type-trunc k B)
        ( Œª h ‚Üí (h ‚àò unit-trunc) ~ (unit-trunc ‚àò f)))
unique-map-trunc {l1} {l2} {A} {B} k f =
  universal-property-trunc k A (trunc k B) (unit-trunc ‚àò f)

map-trunc :
  {l1 l2 : Level} {A : UU l1} {B : UU l2} (k : ùïã) ‚Üí
  (A ‚Üí B) ‚Üí type-trunc k A ‚Üí type-trunc k B
map-trunc k f =
  pr1 (center (unique-map-trunc k f))

Properties

Truncations of homotopic maps are homotopic

naturality-unit-trunc :
  { l1 l2 : Level} {A : UU l1} {B : UU l2} (k : ùïã) (f : A ‚Üí B) ‚Üí
  ( (map-trunc k f) ‚àò unit-trunc) ~ (unit-trunc ‚àò f)
naturality-unit-trunc k f =
  pr2 (center (unique-map-trunc k f))

htpy-uniqueness-map-trunc :
  { l1 l2 : Level} {A : UU l1} {B : UU l2} (k : ùïã) (f : A ‚Üí B) ‚Üí
  ( h : type-trunc k A ‚Üí type-trunc k B) ‚Üí
  ( ( h ‚àò unit-trunc) ~ (unit-trunc ‚àò f)) ‚Üí
  (map-trunc k f) ~ h
htpy-uniqueness-map-trunc k f h H =
  htpy-eq (ap pr1 (contraction (unique-map-trunc k f) (pair h H)))

htpy-trunc :
  {l1 l2 : Level} {A : UU l1} {B : UU l2} {k : ùïã} {f g : A ‚Üí B} ‚Üí
  (f ~ g) ‚Üí (map-trunc k f ~ map-trunc k g)
htpy-trunc {k = k} {f} {g} H =
  htpy-uniqueness-map-trunc
    ( k)
    ( f)
    ( map-trunc k g)
    ( naturality-unit-trunc k g ‚àôh
      inv-htpy (unit-trunc ¬∑l H))

The truncation of the identity map is the identity map

id-map-trunc :
  { l1 : Level} {A : UU l1} (k : ùïã) ‚Üí map-trunc k (id {A = A}) ~ id
id-map-trunc {l1} {A} k =
  htpy-uniqueness-map-trunc k id id refl-htpy

The truncation of a composite is the composite of the truncations

comp-map-trunc :
  { l1 l2 l3 : Level} {A : UU l1} {B : UU l2} {C : UU l3} (k : ùïã)
  ( g : B ‚Üí C) (f : A ‚Üí B) ‚Üí
  ( map-trunc k (g ‚àò f)) ~
  ( (map-trunc k g) ‚àò (map-trunc k f))
comp-map-trunc k g f =
  htpy-uniqueness-map-trunc k
    ( g ‚àò f)
    ( (map-trunc k g) ‚àò (map-trunc k f))
    ( ( (map-trunc k g) ¬∑l (naturality-unit-trunc k f)) ‚àôh
      ( ( naturality-unit-trunc k g) ¬∑r f))

The functorial action of truncations preserves equivalences

module _
  {l1 l2 : Level} {A : UU l1} {B : UU l2} (k : ùïã) (e : A ‚âÉ B)
  where

  map-equiv-trunc : type-trunc k A ‚Üí type-trunc k B
  map-equiv-trunc = map-trunc k (map-equiv e)

  map-inv-equiv-trunc : type-trunc k B ‚Üí type-trunc k A
  map-inv-equiv-trunc = map-trunc k (map-inv-equiv e)

  is-equiv-map-equiv-trunc : is-equiv map-equiv-trunc
  is-equiv-map-equiv-trunc =
    pair
      ( pair
        ( map-inv-equiv-trunc)
        ( inv-htpy (comp-map-trunc k (map-equiv e) (map-inv-equiv e)) ‚àôh
          ( htpy-trunc (issec-map-inv-equiv e) ‚àôh
            id-map-trunc k)))
      ( pair
        ( map-inv-equiv-trunc)
        ( inv-htpy (comp-map-trunc k (map-inv-equiv e) (map-equiv e)) ‚àôh
          ( htpy-trunc (isretr-map-inv-equiv e) ‚àôh
            id-map-trunc k)))

  equiv-trunc : (type-trunc k A ‚âÉ type-trunc k B)
  pr1 equiv-trunc = map-equiv-trunc
  pr2 equiv-trunc = is-equiv-map-equiv-trunc
