Embeddings are closed under homotopies

module _
  {l1 l2 : Level} {A : UU l1} {B : UU l2} (f g : A ‚Üí B) (H : f ~ g)
  where

  abstract
    is-emb-htpy : is-emb g ‚Üí is-emb f
    is-emb-htpy is-emb-g x y =
      is-equiv-top-is-equiv-left-square
        ( ap g)
        ( concat' (f x) (H y))
        ( ap f)
        ( concat (H x) (g y))
        ( nat-htpy H)
        ( is-equiv-concat (H x) (g y))
        ( is-emb-g x y)
        ( is-equiv-concat' (f x) (H y))
