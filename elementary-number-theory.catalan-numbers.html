Idea

The Catalan numbers are a sequence of natural numbers that occur in several combinatorics problems.

Definitions

catalan-numbers : ℕ → ℕ
catalan-numbers =
  strong-ind-ℕ (λ _ → ℕ) (succ-ℕ zero-ℕ)
    ( λ k C →
      sum-Fin-ℕ {k = k}
        ( λ i → mul-ℕ (C (nat-Fin i) (leq-le-ℕ {x = nat-Fin i} (strict-upper-bound-nat-Fin i)))
          ( C (dist-ℕ (nat-Fin i) k) (leq-dist-ℕ (nat-Fin i) k (leq-le-ℕ {x = nat-Fin i} (strict-upper-bound-nat-Fin i))))))

catalan-numbers-binomial : ℕ → ℕ
catalan-numbers-binomial n = dist-ℕ ((mul-ℕ 2 n) choose-ℕ n) ((mul-ℕ 2 n) choose-ℕ (succ-ℕ n))
